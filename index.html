-- حذف تریگرها و جداول قبلی برای ایجاد ساختار جدید بدون خطا DROP TRIGGER IF EXISTS on_auth_user_created ON auth.users; DROP FUNCTION IF EXISTS public.handle_new_user(); DROP TABLE IF EXISTS public.matrix_game_history; DROP TABLE IF EXISTS public.user_word_progress; DROP TABLE IF EXISTS public.user_profiles; -- ۱. جدول مشخصات پیشرفته کاربران (شامل نام کاربری، امتیاز و استریک) CREATE TABLE public.user_profiles ( id UUID REFERENCES auth.users NOT NULL PRIMARY KEY, username TEXT UNIQUE, xp INTEGER DEFAULT 0, streak INTEGER DEFAULT 0, created_at TIMESTAMP WITH TIME ZONE DEFAULT TIMEZONE('utc'::text, NOW()), updated_at TIMESTAMP WITH TIME ZONE DEFAULT TIMEZONE('utc'::text, NOW()) ); -- ۲. جدول پیشرفت کلمات (دقیقاً چه کلماتی یاد گرفته شده یا نشده‌اند) CREATE TABLE public.user_word_progress ( id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, user_id UUID REFERENCES public.user_profiles(id) ON DELETE CASCADE NOT NULL, word_id TEXT NOT NULL, is_learned BOOLEAN DEFAULT FALSE, -- TRUE یعنی یاد گرفته، FALSE یعنی یاد نگرفته updated_at TIMESTAMP WITH TIME ZONE DEFAULT TIMEZONE('utc'::text, NOW()), UNIQUE (user_id, word_id) -- جلوگیری از ثبت کلمه تکراری برای یک کاربر ); -- ۳. جدول تاریخچه بازی ماتریکس (برای اینکه سوالات حل شده دوباره تکرار نشوند) CREATE TABLE public.matrix_game_history ( id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, user_id UUID REFERENCES public.user_profiles(id) ON DELETE CASCADE NOT NULL, word_id TEXT NOT NULL, -- آیدی کلمه‌ای که سوال ماتریکسش حل شده solved_at TIMESTAMP WITH TIME ZONE DEFAULT TIMEZONE('utc'::text, NOW()), UNIQUE (user_id, word_id) ); -- ۴. فعال‌سازی امنیت سطح سطر (RLS) برای هر سه جدول ALTER TABLE public.user_profiles ENABLE ROW LEVEL SECURITY; ALTER TABLE public.user_word_progress ENABLE ROW LEVEL SECURITY; ALTER TABLE public.matrix_game_history ENABLE ROW LEVEL SECURITY; -- ۵. تعریف سیاست‌های امنیتی RLS (هر کاربر فقط به داده‌های خودش دسترسی دارد) CREATE POLICY "Profiles Admin/User Policy" ON public.user_profiles FOR ALL USING (auth.uid() = id); CREATE POLICY "Word Progress Policy" ON public.user_word_progress FOR ALL USING (auth.uid() = user_id); CREATE POLICY "Matrix History Policy" ON public.matrix_game_history FOR ALL USING (auth.uid() = user_id); -- ۶. تابع و تریگر خودکار برای ساخت پروفایل به محض ثبت‌نام کاربر CREATE OR REPLACE FUNCTION public.handle_new_user() RETURNS TRIGGER AS $$ BEGIN INSERT INTO public.user_profiles (id, username, xp, streak) VALUES ( new.id, COALESCE(new.raw_user_meta_data->>'display_username', split_part(new.email, '@', 1)), 0, 0 ); RETURN NEW; END; $$ LANGUAGE plpgsql SECURITY DEFINER; CREATE TRIGGER on_auth_user_created AFTER INSERT ON auth.users FOR EACH ROW EXECUTE PROCEDURE public.handle_new_user();
